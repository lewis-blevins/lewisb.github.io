# Filter columns
df = df.loc[:,df.columns.isin(['Price',
                               'Lot size (m2)',
                               'Living space size (m2)',
                               'Rooms'])]

# Replace values in these columns
df['Rooms'] = df['Rooms'].str.replace('[0-9]* kamers? \(' ,'')
df['Rooms'] = df['Rooms'].str.replace(' slaapkamers?\)' ,'')
df['Living space size (m2)'] = df['Living space size (m2)'].str.replace('m' ,'')
df['Living space size (m2)'] = df['Living space size (m2)'].str.replace('\.', '')
df['Lot size (m2)'] = df['Lot size (m2)'].str.replace('m', '')
df['Lot size (m2)'] = df['Lot size (m2)'].str.replace('\.', '')
df['Price'] = df['Price'].str.replace(' ', '')
df['Price'] = df['Price'].str.replace('\.', '')

# Filter rows
df = df[df.Price != 'Prijs op aanvraag']
df = df[~df['Price'].str.contains('v\.?o\.?n\.?')]
df = df[~df["Rooms"].astype(str).str.match("[0-9]* kamers?")]

# Convert columns to int type

# Rename columns

'Price': 'Price (Euros)',

Remove outliers¶

In [4]:
remove_outliers_from = ['Price (Euros)'
                        ,'Lot size (m2)'
                        ,'Living space size (m2)']

# Remove rows with values 3 standard deviations above/below the mean
for col in remove_outliers_from:
    mean = df[col].mean()
    std_dev = df[col].std()
    lower_cutoff = mean - 3 * std_dev
    upper_cutoff = mean + 3 * std_dev
    df = df[(df[col] > lower_cutoff) & (df[col] < upper_cutoff)]
    

print(str(row_count_with_outliers - df.shape[0]) + " outlier rows removed.")    

295 outlier rows removed.

Train models and test with cross validation¶

In [8]:
X = df[['Lot size (m2)'
        ,'Living space size (m2)'
        ,'Number of bedrooms']]

y = df['Price (Euros)']

kf = KFold(n_splits=10, shuffle=True)
regr = linear_model.LinearRegression()

scores = cross_val_score(regr, X, y, cv=kf)

mean_score = str(round(scores.mean(),3))
min_score = str(round(scores.min(),3))
max_score = str(round(scores.max(),3))
print("Average 10-fold cross validation score: " + mean_score + " (min: " + min_score + ", max: "+ max_score+ ")")
