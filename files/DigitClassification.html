# Design model
model = keras.Sequential([
    keras.layers.Flatten(input_shape=(28, 28)),
    keras.layers.Dense(100, activation='relu'),
    keras.layers.Dense(10, activation='sigmoid')
])

# Compile model
model.compile(optimizer='adam',
              loss='sparse_categorical_crossentropy',
              metrics=['accuracy'])

# Train model
model.fit(X_train, y_train, epochs=10);

image_files = ['number_written_by_hand.jpg', 'number_written_in_ms_paint.jpg'] 

for img_file in image_files:
  image = plt.imread(img_file)
  plt.imshow(image) 
  plt.show()
  image = tf.image.resize(image, [28, 28])
  image = tf.image.adjust_contrast(image, 2) 
  image = tf.image.rgb_to_grayscale(image)
  normalized_image = image / tf.math.reduce_max(image)
  normalized_image = tf.math.abs(1 - normalized_image) # invert pixels
  flattened_normalized_image = tf.reshape(normalized_image, [1, 784]) # flatten image vector
  
  # Make prediction
  prediction = model.predict(flattened_normalized_image)
  prediction_class = np.argmax(prediction[0])
  print("The prediction for file '" + img_file + "' (displayed above) is: " + str(prediction_class))
